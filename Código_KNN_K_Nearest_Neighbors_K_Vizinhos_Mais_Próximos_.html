Exemplo em R

##Vamos usar dados de acidentes de trânsito do DENATRAN ou PRF.

Carregar dados

acidentes <- data.frame(
mes = c("Janeiro", "Fevereiro", "Março", "Abril", "Maio", "Junho", "Julho", "Agosto", "Setembro", "Outubro", "Novembro", "Dezembro"),
total_acidentes = c(90000, 85000, 95000, 92000, 88000, 93000, 89000, 96000, 91000, 94000, 86000, 90000),
mortes = c(1500, 1400, 1600, 1450, 1350, 1550, 1300, 1650, 1500, 1600, 1400, 1450),
classificacao = c("grave", "moderado", "grave", "moderado", "grave", "grave", "moderado", "grave", "moderado", "grave", "moderado", "grave")
)

Dividir dados para treino e teste

set.seed(123)
index <- sample(1:nrow(acidentes), 0.7*nrow(acidentes))
treino <- acidentes[index, ]
teste <- acidentes[-index, ]

Aplicar KNN

library(class)
knn_model <- knn(train = treino[,2:3], test = teste[,2:3], cl = treino$classificacao, k = 1)

Verificar precisão

table(knn_model, teste$classificacao)

Plotar os dados

plot(acidentes$total_acidentes, acidentes$mortes, col = ifelse(acidentes$classificacao == "grave", "red", "blue"),
main = "Acidentes de Trânsito em 2025", xlab = "Total de Acidentes", ylab = "Mortes")
legend("topright", legend = c("Grave", "Moderado"), col = c("red", "blue"), pch = 1)
